package com.einvoicehub.server.domain.mapper;

import com.einvoicehub.server.domain.dto.EinvStoreProviderDto;
import com.einvoicehub.server.domain.entity.EinvStoreProvider;
import org.mapstruct.*;

/**
 * Mapper cho bảng EinvStoreProvider - Cấu hình tích hợp Store Provider
 */
@Mapper(componentModel = "spring", unmappedTargetPolicy = ReportingPolicy.IGNORE)
public interface EinvStoreProviderMapper extends EntityMapper<EinvStoreProviderDto, EinvStoreProvider> {

    @Override
    @Mapping(target = "id", ignore = true)
    @Mapping(target = "createdDate", ignore = true)
    @Mapping(target = "updatedDate", ignore = true)
    EinvStoreProvider toEntity(EinvStoreProviderDto dto);

    @Override
    @BeanMapping(nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
    void partialUpdate(@MappingTarget EinvStoreProvider entity, EinvStoreProviderDto dto);
}
